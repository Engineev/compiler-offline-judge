int cd(int d, string a, string b, string c, int sum) {
    if (d == 1) {
        //println("move " + a + " --> " + c);
        sum++;
    } else {
        sum = cd(d - 1, a, c, b, sum);
        //println("move " + a + " --> " + c);
        sum = cd(d - 1, b, a, c, sum);
        sum++;
    }
    return sum;
}
